### 3) Search 기능을 포함한 Value Help CDS 코드 구현
```abap
Data Definitions - ZVH_CARRIER_B03

@AbapCatalog.viewEnhancementCategory: [#NONE]
@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Value Help Carrid'
@Metadata.ignorePropagatedAnnotations: true
@Search.searchable: true
@ObjectModel.resultSet.sizeCategory: #XS
define view entity ZVH_CARRIER_B03 as select from /DMO/I_Carrier
{
    @Search.defaultSearchElement: true -- SearchBar 검색 대상
    @Search.fuzzinessThreshold: 0.8
    @ObjectModel.text.element: [ 'Name' ]
    @UI.textArrangement: #TEXT_FIRST
    key AirlineID,
    
    @Search.defaultSearchElement: true -- SearchBar 검색 대상
    @Search.fuzzinessThreshold: 0.8
    Name,
    
    @Search.defaultSearchElement: true -- SearchBar 검색 대상
    @Search.fuzzinessThreshold: 0.8
    CurrencyCode
}
```

```abap
Data Definitions - ZDSFLIGHT_CR_B03

@AccessControl.authorizationCheck: #NOT_REQUIRED
@EndUserText.label: 'Consumption View (root, Projection View)'
@Metadata.ignorePropagatedAnnotations: false
@Metadata.allowExtensions: true
@Search.searchable: true
define root view entity ZDSFLIGHT_CR_B03 as projection on ZDSFLIGHT_IR_B03
{
    key Carrid,
    @Search.defaultSearchElement: true
    @Consumption.valueHelpDefinition: [{ 
        entity: { name: '/DMO/I_Connection_StdVH', element: 'ConnectionID' },
        additionalBinding: [{ 
            localElement: 'Carrid', element: 'AirlineID', usage: #FILTER
        }]
    }]
    key Connid,
}
```
