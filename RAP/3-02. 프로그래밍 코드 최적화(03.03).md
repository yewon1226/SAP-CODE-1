### 1) CDS Value Help usage 옵션 기능
- `usage` 는 Value Help에서 필터로 쓸지, 선택값을 되돌려줄지 동작 방식을 정하는 옵션
- 미지정 (= `#FILTER`) → WHERE <VH필드> = <현재값> 로 목록만 필터됨
-  `#RESULT` → <현재필드> = <선택값> 으로 선택 후 자동세팅됨
-  `#FILTER_AND_RESULT` → WHERE … + <현재필드> = <선택값> 둘 다 수행됨
```abap
@Consumption.valueHelpDefinition: [{
    entity: {
        name: '<ValueHelp_Entity_Name>',      // 사용할 Value Help CDS
        element: '<ValueHelp_Key_Field>'      // 선택 시 <현재필드>에 들어갈 Key
    },
    additionalBinding: [{
        localElement: '<현재뷰_필드>',         // 현재 Projection View 필드
        element: '<ValueHelp_필드>',          // Value Help CDS 필드
        usage: #FILTER                       // WHERE <ValueHelp_필드> = <현재뷰_필드값>
        // usage: #RESULT                    // <현재뷰_필드> = <선택한 ValueHelp_필드값>
        // usage: #FILTER_AND_RESULT          // WHERE 조건 + 자동세팅 둘 다
        // 미지정 = 기본 #FILTER
    }]
}]
<FieldName>                                   // Value Help 적용 대상 필드
```
```abap
// Data Definitions - ZDSFLIGHT_CR_B03

define root view entity ZDSFLIGHT_CR_B03 as projection on ZDSFLIGHT_IR_B03
{
    key Carrid,
    @Search.defaultSearchElement: true
    @Consumption.valueHelpDefinition: [{ 
        entity: { name: '/DMO/I_Connection_StdVH', element: 'ConnectionID' },
        additionalBinding: [{ 
            localElement: 'Carrid', element: 'AirlineID', usage: #FILTER
        }]
    }]
    key Connid,
}
```

### 2) CDS ResultSet 크기 설정과 UI 표시 방식
- 조회 결과 데이터 규모를 시스템에 알려주는 힌트이며, 값에 따라 Value Help UI 방식이 달라질 수 있음
```abap
@ObjectModel.resultSet.sizeCategory: #XS   // 매우 적은 데이터 (전체 로딩 가능 → 드롭다운 가능)
@ObjectModel.resultSet.sizeCategory: #S    // 적은 데이터
@ObjectModel.resultSet.sizeCategory: #M    // 중간 규모 데이터
@ObjectModel.resultSet.sizeCategory: #L    // 많은 데이터
@ObjectModel.resultSet.sizeCategory: #XL   // 매우 많은 데이터 (검색 팝업 형태)

@ObjectModel.resultSet.sizeCategory: #XS
define view entity <ValueHelp_View> as select from <Source_Table>
{
    key <Key_Field>,
        <Text_Field>
}
```
```abap
// Data Definitions - ZVH_CARRIER_B03

@Search.searchable: true
@ObjectModel.resultSet.sizeCategory: #XS   // #XS / #S / #M / #L / #XL 가능

    key AirlineID,
        Name,
```

### 4) `@Consumption.filter` 적용 위치 (Projection View에서도 가능)
- `@Consumption.filter` 는 메타데이터 확장뷰뿐 아니라 Projection View 안에서도 직접 설정 가능
```abap
define root view entity <Projection_View> as projection on <Interface_View>
{
  @Consumption.filter: {
      selectionType: #SINGLE,     // #SINGLE / #MULTIPLE
      defaultValue: '<기본값>'
  }
  <FieldName>
}
```
```abap
// Data Definitions - ZDSFLIGHT_CR_B03

define root view entity ZDSFLIGHT_CR_B03 as projection on ZDSFLIGHT_IR_B03
{
  @Consumption.filter: {
      selectionType: #SINGLE,
      defaultValue: 'AZ'
  }
  key Carrid
}
```

### 5) CDS `@UI.presentationVariant` 정렬·표시 기본설정
- List Report가 처음 열릴 때 적용될 기본 정렬·표시 방식을 설정하는 옵션
```abap
@UI: {
  presentationVariant: [{
      sortOrder: [
          { by: '<Field1>', direction: #ASC },   // 오름차순
          { by: '<Field2>', direction: #DESC }   // 내림차순
      ]
  }]
}
```
```abap
// Metadata Extensions - ZDSFLIGHT_CR_B03

}
annotate view ZDSFLIGHT_CR_B03 with { ... }
```

### 6) CDS `@UI.facet` – Object Page 섹션 구성 설정 
- Object Page의 섹션(블록) 구조를 정의하는 레이아웃 설정이며, `type` 값에 따라 표시 방식이 달라짐
```abap
type: #IDENTIFICATION_REFERENCE   // 상단 기본정보 영역 (@UI.identification 연결)
type: #FIELDGROUP_REFERENCE       // 필드 묶음 영역 (@UI.fieldGroup 연결)
type: #LINEITEM_REFERENCE         // 하위 테이블 영역 (Association 연결)
type: #CHART_REFERENCE            // 차트 영역 (@UI.chart 연결)
type: #DATAPOINT_REFERENCE        // KPI 단일 값 영역 (@UI.dataPoint 연결)
type: #CONTACT_REFERENCE          // 연락처 정보 영역 (@UI.contact 연결)
type: #COLLECTION                 // 여러 facet을 묶는 그룹(컨테이너)

purpose: #STANDARD   // 일반 섹션 (본문 영역)
purpose: #HEADER     // Object Page 상단 Header 영역
```
```abap
@UI.facet: [
  {
    id: '<Facet_ID>',                 // 섹션 고유 ID
    purpose: #STANDARD,               // HEADER / STANDARD
    type: #LINEITEM_REFERENCE,        // 어떤 UI 블록인지 결정
    label: '<섹션명>',                // 화면에 표시될 제목
    position: 10,                     // 화면 표시 순서
    targetElement: '<Association>',   // 하위 테이블 연결 시 사용
    targetQualifier: '<Qualifier>'    // identification/fieldGroup 연결 시 사용
  }
]
```
